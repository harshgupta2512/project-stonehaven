// -- Text scramble effect --
function scrambleText(el) {
  var finalText = el.dataset.text;
  if (!finalText) return;
  var chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ_-/\\';
  var iterations = 0;
  var maxIterations = finalText.length;
  var interval = setInterval(function() {
    el.textContent = finalText.split('').map(function(c, i) {
      if (i < iterations) return finalText[i];
      return chars[Math.floor(Math.random() * chars.length)];
    }).join('');
    iterations += 0.6;
    if (iterations >= maxIterations) {
      el.textContent = finalText;
      clearInterval(interval);
    }
  }, 35);
}

function animateCounters() {
  document.querySelectorAll('[data-target]').forEach(function(el) {
    var target = parseFloat(el.dataset.target);
    var type = el.dataset.type;
    var suffix = el.dataset.suffix || '';
    var duration = 2400;
    var start = performance.now();
    function update(now) {
      var progress = Math.min((now - start) / duration, 1);
      // Elastic ease out
      var eased = progress === 1 ? 1 : 1 - Math.pow(2, -10 * progress) * Math.cos((progress * 10 - 0.75) * (2 * Math.PI) / 3);
      eased = Math.max(0, Math.min(1, eased));
      if (type === 'price') {
        el.innerHTML = '$' + (target * eased).toFixed(1) + '<span class="spec-unit">M+</span>';
      } else {
        el.innerHTML = Math.round(target * eased) + (suffix ? '<span class="spec-unit">' + suffix + '</span>' : '');
      }
      if (progress < 1) requestAnimationFrame(update);
    }
    requestAnimationFrame(update);
  });
}

(function() {
  var canvas = document.getElementById('heroOcean');
  var ctx = canvas.getContext('2d');
  var waveHeight = isSmallScreen ? 140 : 220;
  var waveStep = isSmallScreen ? 4 : 2;
  function resize() { canvas.width = window.innerWidth; canvas.height = waveHeight; }
  resize();
  window.addEventListener('resize', resize);

  function drawWave(t) {
    ctx.clearRect(0, 0, canvas.width, canvas.height);
    // Wave 1 - largest
    ctx.beginPath();
    ctx.moveTo(0, canvas.height);
    for (var x = 0; x <= canvas.width; x += waveStep) {
      var y = Math.sin(x * 0.003 + t * 0.7) * 22 + Math.sin(x * 0.007 + t * 0.4) * 12 + 75;
      ctx.lineTo(x, y);
    }
    ctx.lineTo(canvas.width, canvas.height);
    ctx.fillStyle = 'rgba(111,168,199,0.07)';
    ctx.fill();
    // Wave 2
    ctx.beginPath();
    ctx.moveTo(0, canvas.height);
    for (var x2 = 0; x2 <= canvas.width; x2 += waveStep) {
      var y2 = Math.sin(x2 * 0.004 + t * 0.55 + 2) * 16 + Math.sin(x2 * 0.009 + t * 0.28) * 9 + 105;
      ctx.lineTo(x2, y2);
    }
    ctx.lineTo(canvas.width, canvas.height);
    ctx.fillStyle = 'rgba(196,169,125,0.05)';
    ctx.fill();
    // Wave 3 - smallest
    ctx.beginPath();
    ctx.moveTo(0, canvas.height);
    for (var x3 = 0; x3 <= canvas.width; x3 += waveStep) {
      var y3 = Math.sin(x3 * 0.005 + t * 0.9 + 4) * 10 + Math.sin(x3 * 0.012 + t * 0.35) * 6 + 140;
      ctx.lineTo(x3, y3);
    }
    ctx.lineTo(canvas.width, canvas.height);
    ctx.fillStyle = 'rgba(111,168,199,0.035)';
    ctx.fill();
    // Wave 4 - accent
    ctx.beginPath();
    ctx.moveTo(0, canvas.height);
    for (var x4 = 0; x4 <= canvas.width; x4 += (waveStep + 1)) {
      var y4 = Math.sin(x4 * 0.006 + t * 1.1 + 1) * 7 + Math.sin(x4 * 0.014 + t * 0.5) * 4 + 170;
      ctx.lineTo(x4, y4);
    }
    ctx.lineTo(canvas.width, canvas.height);
    ctx.fillStyle = 'rgba(196,169,125,0.025)';
    ctx.fill();
  }

  var t = 0;
  function animate() {
    t += 0.018;
    drawWave(t);
    requestAnimationFrame(animate);
  }
  animate();
})();

      var formEl = document.getElementById('contactForm');
      var successEl = document.getElementById('formSuccess');
      if (formEl && successEl) {
        formEl.style.display = 'none';
        successEl.classList.add('active');
      } else {

        toast.classList.add('show');
        setTimeout(function() { toast.classList.remove('show'); }, 4000);
      }
      form.reset();
      document.querySelectorAll('.form-group').forEach(function(g) { g.classList.remove('has-value'); });
      btn.textContent = origText;
    } else {
      throw new Error('Form submission failed');
    }
  })
  .catch(function() {
    btn.textContent = origText;
    alert('Something went wrong. Please try again or contact us directly at hello@kaizenprojects.com.au');
  });

// -- Interactive Map Tooltips --
(function() {
  var mapContainer = document.getElementById('interactiveMap');
  if (!mapContainer) return;
  var tooltip = document.getElementById('mapTooltip');
  if (!tooltip) return;
  var tooltipName = tooltip.querySelector('.map-tooltip-name');
  var tooltipDetail = tooltip.querySelector('.map-tooltip-detail');
  var tooltipDist = tooltip.querySelector('.map-tooltip-distance');
  var markers = mapContainer.querySelectorAll('.map-marker');

  markers.forEach(function(marker) {
    marker.addEventListener('mouseenter', function(e) {
      var name = marker.getAttribute('data-name');
      var desc = marker.getAttribute('data-desc');
      var dist = marker.getAttribute('data-dist');
      if (!name) return;
      tooltipName.textContent = name;
      tooltipDetail.textContent = desc || '';
      tooltipDist.textContent = dist || '';

      // Position tooltip above the marker
      var cx = parseFloat(marker.getAttribute('data-cx'));
      var cy = parseFloat(marker.getAttribute('data-cy'));
      var svg = mapContainer.querySelector('svg');
      var svgRect = svg.getBoundingClientRect();
      var scaleX = svgRect.width / 800;
      var scaleY = svgRect.height / 600;
      var tooltipX = cx * scaleX;
      var tooltipY = cy * scaleY;

      tooltip.style.left = tooltipX + 'px';
      tooltip.style.top = (tooltipY - 20) + 'px';
      tooltip.style.transform = 'translate(-50%, -100%)';
      tooltip.classList.add('active');
    });

    marker.addEventListener('mouseleave', function() {
      tooltip.classList.remove('active');
    });
  });

  // Touch support
  if (isTouchDevice) {
    var activeMarker = null;
    markers.forEach(function(marker) {
      marker.addEventListener('click', function(e) {
        e.preventDefault();
        e.stopPropagation();
        if (activeMarker === marker) {
          tooltip.classList.remove('active');
          activeMarker = null;
          return;
        }
        activeMarker = marker;
        marker.dispatchEvent(new Event('mouseenter'));
      });
    });
    mapContainer.addEventListener('click', function(e) {
      if (!e.target.closest('.map-marker')) {
        tooltip.classList.remove('active');
        activeMarker = null;
      }
    });
  }
})();

(function() {
  var overlay  = document.getElementById('lightbox');
  var img      = document.getElementById('lightboxImg');
  var label    = document.getElementById('lightboxLabel');
  var title    = document.getElementById('lightboxTitle');
  var counter  = document.getElementById('lightboxCounter');
  var closeBtn = document.getElementById('lightboxClose');
  var prevBtn  = document.getElementById('lightboxPrev');
  var nextBtn  = document.getElementById('lightboxNext');

  // Build gallery items from gallery cards
  var items = [];
  document.querySelectorAll('.gallery-card').forEach(function(card) {
    var cardImg = card.querySelector('img');
    var cardLabel = card.querySelector('.gallery-card-label');
    var cardTitle = card.querySelector('.gallery-card-title');
    if (cardImg) {
      items.push({
        src: cardImg.src,
        label: cardLabel ? cardLabel.textContent : '',
        title: cardTitle ? cardTitle.textContent : ''
      });
    }
  });

  var current = 0;

  function show(index) {
    if (index < 0) index = items.length - 1;
    if (index >= items.length) index = 0;
    current = index;

    // Brief image transition
    img.style.opacity = '0';
    img.style.transform = 'scale(0.97)';
    setTimeout(function() {
      img.src = items[current].src;
      img.alt = items[current].title;
      label.textContent = items[current].label;
      title.textContent = items[current].title;
      counter.textContent = (current + 1) + ' / ' + items.length;
      img.style.opacity = '1';
      img.style.transform = 'scale(1)';
    }, 150);
  }

  function open(index) {
    show(index);
    overlay.classList.add('active');
    document.body.style.overflow = 'hidden';
    img.style.transition = 'opacity 0.3s, transform 0.4s cubic-bezier(0.16,1,0.3,1)';
  }

  function close() {
    overlay.classList.remove('active');
    document.body.style.overflow = '';
  }

  // Click gallery cards to open lightbox
  document.querySelectorAll('.gallery-card').forEach(function(card, i) {
    card.style.cursor = 'pointer';
    card.addEventListener('click', function(e) {
      // Don't open lightbox on drag
      if (Math.abs(card._dragDelta || 0) > 5) return;
      open(i);
    });

    // Track drag distance to prevent opening on drag end
    card.addEventListener('mousedown', function(e) { card._startX = e.pageX; });
    card.addEventListener('mouseup', function(e) { card._dragDelta = Math.abs(e.pageX - (card._startX || 0)); });
    card.addEventListener('touchstart', function(e) { card._startX = e.touches[0].pageX; });
    card.addEventListener('touchend', function(e) { card._dragDelta = Math.abs((e.changedTouches[0].pageX) - (card._startX || 0)); });
  });

  closeBtn.addEventListener('click', close);
  prevBtn.addEventListener('click', function(e) { e.stopPropagation(); show(current - 1); });
  nextBtn.addEventListener('click', function(e) { e.stopPropagation(); show(current + 1); });

  // Click backdrop to close
  overlay.addEventListener('click', function(e) {
    if (e.target === overlay) close();
  });

  // Keyboard nav
  document.addEventListener('keydown', function(e) {
    if (!overlay.classList.contains('active')) return;
    if (e.key === 'Escape') close();
    if (e.key === 'ArrowLeft')  show(current - 1);
    if (e.key === 'ArrowRight') show(current + 1);
  });

  // Touch swipe
  var touchStartX = 0;
  overlay.addEventListener('touchstart', function(e) { touchStartX = e.changedTouches[0].screenX; });
  overlay.addEventListener('touchend', function(e) {
    var diff = touchStartX - e.changedTouches[0].screenX;
    if (Math.abs(diff) > 50) {
      if (diff > 0) show(current + 1);
      else show(current - 1);
    }
  });
})();

// -- FLOOR PLAN INTERACTIVE VIEWER --
(function() {
  var viewer = document.getElementById('fpD1');
  var img = document.getElementById('fpImg');
  if (!viewer || !img) return;

  var scale = 0.45, rotation = 0, panX = 0, panY = 0;
  var minScale = 0.25, maxScale = 3;
  var isDragging = false, startX = 0, startY = 0, startPanX = 0, startPanY = 0;

  function updateTransform() {
    img.style.transform = 'translate(-50%,-50%) translate(' + panX + 'px,' + panY + 'px) scale(' + scale + ') rotate(' + rotation + 'deg)';
  }

  function fitToView() {
    var vw = viewer.clientWidth, vh = viewer.clientHeight;
    var iw = img.naturalWidth || 1526, ih = img.naturalHeight || 2879;
    var scaleW = vw / iw, scaleH = vh / ih;
    scale = Math.min(scaleW, scaleH) * 0.9;
    panX = 0; panY = 0;
    updateTransform();
  }

  img.onload = fitToView;
  if (img.complete) fitToView();

  // Zoom buttons
  document.getElementById('fpZoomIn').addEventListener('click', function() {
    scale = Math.min(scale * 1.3, maxScale); updateTransform();
  });
  document.getElementById('fpZoomOut').addEventListener('click', function() {
    scale = Math.max(scale / 1.3, minScale); updateTransform();
  });
  document.getElementById('fpRotate').addEventListener('click', function() {
    rotation = (rotation + 90) % 360; updateTransform();
  });
  document.getElementById('fpReset').addEventListener('click', function() {
    rotation = 0; fitToView();
  });

  // Mouse drag
  viewer.addEventListener('mousedown', function(e) {
    if (e.target.closest('.fp-controls')) return;
    isDragging = true; startX = e.clientX; startY = e.clientY;
    startPanX = panX; startPanY = panY;
    e.preventDefault();
  });
  window.addEventListener('mousemove', function(e) {
    if (!isDragging) return;
    panX = startPanX + (e.clientX - startX);
    panY = startPanY + (e.clientY - startY);
    updateTransform();
  });
  window.addEventListener('mouseup', function() { isDragging = false; });

  // Scroll wheel zoom
  viewer.addEventListener('wheel', function(e) {
    e.preventDefault();
    var delta = e.deltaY > 0 ? 0.9 : 1.1;
    scale = Math.min(Math.max(scale * delta, minScale), maxScale);
    updateTransform();
  }, { passive: false });

  // Touch: pinch zoom + pan
  var touches0 = null, touchDist0 = 0, touchScale0 = 1;
  viewer.addEventListener('touchstart', function(e) {
    if (e.target.closest('.fp-controls')) return;
    if (e.touches.length === 1) {
      isDragging = true; startX = e.touches[0].clientX; startY = e.touches[0].clientY;
      startPanX = panX; startPanY = panY;
    } else if (e.touches.length === 2) {
      isDragging = false;
      var dx = e.touches[0].clientX - e.touches[1].clientX;
      var dy = e.touches[0].clientY - e.touches[1].clientY;
      touchDist0 = Math.sqrt(dx * dx + dy * dy);
      touchScale0 = scale;
    }
  }, { passive: true });
  viewer.addEventListener('touchmove', function(e) {
    if (e.touches.length === 1 && isDragging) {
      panX = startPanX + (e.touches[0].clientX - startX);
      panY = startPanY + (e.touches[0].clientY - startY);
      updateTransform();
      e.preventDefault();
    } else if (e.touches.length === 2) {
      var dx = e.touches[0].clientX - e.touches[1].clientX;
      var dy = e.touches[0].clientY - e.touches[1].clientY;
      var dist = Math.sqrt(dx * dx + dy * dy);
      scale = Math.min(Math.max(touchScale0 * (dist / touchDist0), minScale), maxScale);
      updateTransform();
      e.preventDefault();
    }
  }, { passive: false });
  viewer.addEventListener('touchend', function() { isDragging = false; });
})();

// -- COMPARISON SLIDER --
(function() {
  var slides = document.querySelectorAll('.compare-slide');
  var dotsWrap = document.getElementById('compareDots');
  var prevBtn = document.getElementById('comparePrev');
  var nextBtn = document.getElementById('compareNext');
  var roomTabs = document.querySelectorAll('.compare-room-tab');
  if (!slides.length) return;

  var current = 0;

  // Build dots
  slides.forEach(function(_, i) {
    var dot = document.createElement('div');
    dot.className = 'compare-dot' + (i === 0 ? ' active' : '');
    dot.addEventListener('click', function() { goTo(i); });
    dotsWrap.appendChild(dot);
  });

  function goTo(index) {
    slides[current].classList.remove('active');
    dotsWrap.children[current].classList.remove('active');
    current = (index + slides.length) % slides.length;
    slides[current].classList.add('active');
    dotsWrap.children[current].classList.add('active');
    // Update room tabs
    roomTabs.forEach(function(tab) { tab.classList.remove('active'); });
    if (roomTabs[current]) roomTabs[current].classList.add('active');
    // Reset slider to 50%
    var box = slides[current].querySelector('.compare-container');
    if (box) setSlider(box, 0.5);
  }
  prevBtn.addEventListener('click', function() { goTo(current - 1); });
  nextBtn.addEventListener('click', function() { goTo(current + 1); });

  // Room tab click handlers
  roomTabs.forEach(function(tab) {
    tab.addEventListener('click', function() {
      var room = parseInt(tab.getAttribute('data-room'));
      goTo(room);
    });
  });

  // Slider logic for each compare-container
  function setSlider(container, pct) {
    pct = Math.max(0.02, Math.min(0.98, pct));
    var afterImg = container.querySelector('.compare-img--after');
    var divider = container.querySelector('.compare-divider');
    var handle = container.querySelector('.compare-handle');
    var labelBefore = container.querySelector('.compare-label--before');
    var labelAfter = container.querySelector('.compare-label--after');
    var percent = pct * 100 + '%';
    afterImg.style.clipPath = 'inset(0 0 0 ' + percent + ')';
    divider.style.left = percent;
    handle.style.left = percent;
    // Fade labels near edges
    if (labelBefore) labelBefore.style.opacity = pct > 0.15 ? 1 : pct / 0.15;
    if (labelAfter) labelAfter.style.opacity = pct < 0.85 ? 1 : (1 - pct) / 0.15;
  }

  document.querySelectorAll('.compare-container').forEach(function(container) {
    var dragging = false;

    function getPercent(e) {
      var rect = container.getBoundingClientRect();
      var clientX = e.touches ? e.touches[0].clientX : e.clientX;
      return (clientX - rect.left) / rect.width;
    }

    function onStart(e) {
      e.preventDefault();
      dragging = true;
      container.classList.add('dragging');
      setSlider(container, getPercent(e));
    }
    function onMove(e) {
      if (!dragging) return;
      e.preventDefault();
      setSlider(container, getPercent(e));
    }
    function onEnd() {
      dragging = false;
      container.classList.remove('dragging');
    }

    container.addEventListener('mousedown', onStart);
    container.addEventListener('touchstart', onStart, { passive: false });
    window.addEventListener('mousemove', onMove);
    window.addEventListener('touchmove', onMove, { passive: false });
    window.addEventListener('mouseup', onEnd);
    window.addEventListener('touchend', onEnd);

    // Intro animation: slide from 0 to 50% on first view
    var animated = false;
    var observer = new IntersectionObserver(function(entries) {
      entries.forEach(function(entry) {
        if (entry.isIntersecting && !animated) {
          animated = true;
          var start = null;
          function animate(ts) {
            if (!start) start = ts;
            var progress = Math.min((ts - start) / 1200, 1);
            var ease = 1 - Math.pow(1 - progress, 4); // easeOutQuart
            setSlider(container, ease * 0.5);
            if (progress < 1) requestAnimationFrame(animate);
          }
          requestAnimationFrame(animate);
        }
      });
    }, { threshold: 0.3 });
    observer.observe(container);
  });
})();

// -- Popup Lead Form for All Visitors --
(function() {
  // Don't show if already submitted or dismissed this session
  if (sessionStorage.getItem('riva_popup_dismissed')) return;

  var overlay = document.getElementById('popupOverlay');
  if (!overlay) return;

  // Show popup after 4 seconds — enough to see the hero, not so long they've scrolled away
  setTimeout(function() {
    // Don't show if user has already scrolled to the contact form
    var contactSection = document.getElementById('contact');
    if (contactSection) {
      var rect = contactSection.getBoundingClientRect();
      if (rect.top < window.innerHeight) return;
    }
    overlay.classList.add('active');
    document.body.style.overflow = 'hidden';
  }, 4000);

  // Close handlers
  var closeBtn = overlay.querySelector('.popup-close');
  closeBtn.addEventListener('click', closePopup);
  overlay.addEventListener('click', function(e) {
    if (e.target === overlay) closePopup();
  });
  document.addEventListener('keydown', function(e) {
    if (e.key === 'Escape' && overlay.classList.contains('active')) closePopup();
  });

  function closePopup() {
    overlay.classList.remove('active');
    document.body.style.overflow = '';
    sessionStorage.setItem('riva_popup_dismissed', '1');
  }

  // Popup form interactions
  var popupForm = document.getElementById('popupForm');
  if (!popupForm) return;

  // Floating labels for popup form
  popupForm.querySelectorAll('.form-group input, .form-group textarea').forEach(function(input) {
    input.addEventListener('input', function() {
      input.closest('.form-group').classList.toggle('has-value', input.value.length > 0);
    });
  });

  // Copy UTM values into popup form hidden fields
  var params = new URLSearchParams(window.location.search);
  var utmFields = ['utm_source', 'utm_medium', 'utm_campaign', 'utm_content', 'utm_term'];
  utmFields.forEach(function(f) {
    var val = params.get(f);
    if (val) {
      var el = popupForm.querySelector('input[name="' + f + '"]');
      if (el) el.value = val;
    }
  });
  var popupRef = popupForm.querySelector('input[name="referrer"]');
  if (popupRef) popupRef.value = document.referrer || 'direct';
  var popupLp = popupForm.querySelector('input[name="landing_page"]');
  if (popupLp) popupLp.value = window.location.href;

  popupForm.addEventListener('submit', function(e) {
    e.preventDefault();
    var btn = popupForm.querySelector('button[type="submit"] span');
    var origText = btn.textContent;
    btn.textContent = 'SENDING...';

    fetch('/', {
      method: 'POST',
      headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
      body: new URLSearchParams(new FormData(popupForm)).toString()
    })
    .then(function(res) {
      if (res.ok) {
        if (typeof fbq === 'function') { fbq('track', 'Lead', { content_name: 'Riva Floor Plan Request - Popup' }); }
        if (typeof gtag === 'function') { gtag('event', 'conversion', { send_to: 'AW-XXXXX/YYYYY' }); }
        // Hide form content, show success
        overlay.querySelector('.popup-header').style.display = 'none';
        overlay.querySelector('.popup-body').style.display = 'none';
        document.getElementById('popupSuccess').classList.add('active');
        popupForm.reset();
        sessionStorage.setItem('riva_popup_dismissed', '1');
        // Auto-close after 5 seconds
        setTimeout(function() { closePopup(); }, 5000);
      } else {
        throw new Error('Failed');
      }
    })
    .catch(function() {
      btn.textContent = origText;
      alert('Something went wrong. Please try again or call Troy on 0449 763 628.');
    });
  });
})();
